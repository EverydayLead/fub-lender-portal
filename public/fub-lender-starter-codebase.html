i

Prerequisites

  • Node.js 18+ and npm 9+
  • Expo CLI (npm install -g expo-cli)
  • EAS CLI (npm install -g eas-cli)
  • Xcode 15+ with iOS Simulator (for iOS development)
  • Android Studio + Android SDK 34 (for Android development)
  • CocoaPods (sudo gem install cocoapods)
1

Clone & Install

Pull the repo and install all dependencies.

git clone git@github.com:first-united/fub-lender.git cd fub-lender && npm install cd ios && pod install && cd ..
2

Configure Environment

Copy the env template and fill in your values.

cp .env.example .env.local # Edit .env.local with your Okta + API values
3

Run in Demo Mode

Start the app with mock data — no backend needed.

DEMO_MODE=true npx expo run:ios # or: DEMO_MODE=true npx expo run:android
4

Run Against Backend

Point at a real or staging API server.

npx expo run:ios # Ensure API_BASE_URL is set in .env.local

!

Why Expo Custom Dev Client (not Expo Go)?

WatermelonDB requires native SQLite bindings, and the Okta React Native SDK uses native OIDC flows. Both need custom native modules that Expo Go cannot load. The project uses expo prebuild to generate the ios/ and android/ directories with the correct native linking.

✓

Schema ↔ PostgreSQL Alignment

Every WatermelonDB table maps directly to a PostgreSQL table from Prompt 5. The server_id column on each local table references the UUID primary key on the server. The authority_matrix table from PG is not replicated locally — it's fetched on demand via the API and cached in memory via TanStack Query, since authority levels rarely change.

Network Layer
API Client & Sync Engine
Axios with automatic Okta token injection, exponential retry, and an offline queue that persists mutations until connectivity returns.
🌐 src/api/client.ts TypeScript
import axios, { AxiosError, InternalAxiosRequestConfig } from 'axios'; import * as Network from 'expo-network'; import { useAuthStore } from '../store/authStore'; import { useSyncStore } from '../store/syncStore'; const API_BASE = process.env.API_BASE_URL ?? 'https://lender-api.firstunitedbank.com/api'; const TIMEOUT = Number(process.env.API_TIMEOUT_MS) || 15000; const MAX_RETRIES = 3; // ── Create Axios Instance ── export const apiClient = axios.create({ baseURL: API_BASE, timeout: TIMEOUT, headers: { 'Content-Type': 'application/json' }, }); // ── Request Interceptor: Inject Okta Bearer Token ── apiClient.interceptors.request.use( async (config: InternalAxiosRequestConfig) => { const { accessToken } = useAuthStore.getState(); if (accessToken) { config.headers.Authorization = `Bearer ${accessToken}`; } // Tag request with connectivity state for retry logic const network = await Network.getNetworkStateAsync(); (config as any).__isOnline = network.isConnected && network.isInternetReachable; return config; }, (error) => Promise.reject(error), ); // ── Response Interceptor: Retry + Offline Queue ── apiClient.interceptors.response.use( (response) => response, async (error: AxiosError) => { const config = error.config as any; if (!config) return Promise.reject(error); // ── Token Expired: Attempt refresh ── if (error.response?.status === 401 && !config.__retryAuth) { config.__retryAuth = true; try { await useAuthStore.getState().refreshToken(); const { accessToken } = useAuthStore.getState(); config.headers.Authorization = `Bearer ${accessToken}`; return apiClient(config); } catch { useAuthStore.getState().logout(); return Promise.reject(error); } } // ── Network Error or 5xx: Exponential Backoff Retry ── const retryCount = config.__retryCount ?? 0; const isRetryable = !error.response || (error.response.status >= 500 && error.response.status < 600); if (isRetryable && retryCount < MAX_RETRIES) { config.__retryCount = retryCount + 1; const delay = Math.pow(2, retryCount) * 1000; await new Promise((r) => setTimeout(r, delay)); return apiClient(config); } // ── Offline: Queue mutation for later ── if (!config.__isOnline && ['post', 'put', 'patch', 'delete'].includes(config.method)) { useSyncStore.getState().enqueue({ method: config.method, url: config.url, data: config.data, timestamp: Date.now(), }); return { data: null, __queued: true } as any; } return Promise.reject(error); }, );

import * as Network from 'expo-network'; import { database } from '../db/database'; import { apiClient } from '../api/client'; import { useSyncStore } from '../store/syncStore'; interface SyncQueueItem { id: string; method: string; url: string; data: any; timestamp: number; retryCount: number; } const MAX_RETRY = 5; const SYNC_INTERVAL = Number(process.env.SYNC_INTERVAL_MS) || 30000; export class SyncManager { private intervalId: ReturnType<typeof setInterval> | null = null; private isSyncing = false; /** Start the periodic sync loop */ start() { if (this.intervalId) return; this.intervalId = setInterval(() => this.sync(), SYNC_INTERVAL); this.sync(); // Run immediately on start } /** Stop the sync loop */ stop() { if (this.intervalId) { clearInterval(this.intervalId); this.intervalId = null; } } /** Execute a full sync cycle: push local changes, then pull remote */ async sync(): Promise<void> { if (this.isSyncing) return; const network = await Network.getNetworkStateAsync(); if (!network.isConnected || !network.isInternetReachable) { useSyncStore.getState().setStatus('offline'); return; } this.isSyncing = true; useSyncStore.getState().setStatus('syncing'); try { // Phase 1: Push — flush the offline queue await this.pushLocalChanges(); // Phase 2: Pull — fetch changes since last sync await this.pullRemoteChanges(); useSyncStore.getState().setStatus('synced'); useSyncStore.getState().setLastSyncAt(Date.now()); } catch (err) { console.error('[SyncManager] Sync failed:', err); useSyncStore.getState().setStatus('error'); } finally { this.isSyncing = false; } } /** Push queued mutations to the server */ private async pushLocalChanges(): Promise<void> { const queue = useSyncStore.getState().queue; const pending = queue.filter((item) => item.retryCount < MAX_RETRY); for (const item of pending) { try { await apiClient({ method: item.method, url: item.url, data: item.data, }); useSyncStore.getState().dequeue(item.id); } catch (err) { useSyncStore.getState().incrementRetry(item.id); console.warn(`[Sync] Failed to push ${item.id}, retry ${item.retryCount + 1}`); } } } /** Pull changes from server since last sync timestamp */ private async pullRemoteChanges(): Promise<void> { const lastSync = useSyncStore.getState().lastSyncAt; const since = lastSync ? new Date(lastSync).toISOString() : undefined; const { data } = await apiClient.get('/sync/pull', { params: { since }, }); if (!data?.changes) return; await database.write(async () => { for (const change of data.changes) { const collection = database.get(change.table); switch (change.operation) { case 'create': await collection.create((record: any) => { Object.assign(record, change.data); }); break; case 'update': const existing = await collection.find(change.localId); await existing.update((r: any) => Object.assign(r, change.data)); break; case 'delete': const toDelete = await collection.find(change.localId); await toDelete.markAsDeleted(); break; } } }); } } export const syncManager = new SyncManager();

import { useAuthStore } from '../store/authStore'; /** * FUB Lending Authority Matrix * Individual lending authorities in the authority_matrix table * determine whether a user can approve a deal at a given amount. */ export interface AuthorityCheck { canApprove: boolean; reason?: string; requiredLevel?: number; escalateTo?: string; // role title of the next approver } /** Check if the current user can approve a deal of the given amount */ export function checkAuthority(dealAmount: number): AuthorityCheck { const user = useAuthStore.getState().user; if (!user) { return { canApprove: false, reason: 'Not authenticated' }; } if (dealAmount <= user.authorityLevel) { return { canApprove: true }; } // Determine who needs to approve based on amount thresholds const escalation = getEscalationTarget(dealAmount); return { canApprove: false, reason: `Amount exceeds your authority ($${(user.authorityLevel / 1000).toFixed(0)}K limit)`, requiredLevel: dealAmount, escalateTo: escalation, }; } /** Determine the approval chain based on deal amount */ function getEscalationTarget(amount: number): string { if (amount <= 250_000) return 'Community Bank President'; if (amount <= 1_000_000) return 'Regional Credit Officer'; if (amount <= 5_000_000) return 'Chief Credit Officer'; return 'Loan Committee'; } /** Get SLA hours based on the deal's complexity lane */ export function getSLAHours(lane: 'fast' | 'standard' | 'complex'): number { switch (lane) { case 'fast': return 4; // 4 business hours case 'standard': return 24; // 3 business days case 'complex': return 72; // 9 business days } }
